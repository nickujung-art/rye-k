// 복식부기 엔진 — 순수 함수 (Supabase 의존성 없음, 테스트 용이)

export interface LineInput {
  account_id: string
  side: 'debit' | 'credit'
  amount: number
  memo?: string
}

export interface EntryInput {
  tenant_id: string
  entry_date: string          // YYYY-MM-DD
  description: string
  source_type?: 'ryek_payment' | 'bank_import' | 'payroll' | 'manual' | 'invoice'
  source_id?: string
  lines: LineInput[]
}

// 차대변 균형 검증 (DB 트리거와 동일 로직 — 클라이언트 사전 검증용)
export function validateBalance(lines: LineInput[]): { valid: boolean; debit: number; credit: number; diff: number } {
  const debit  = lines.filter(l => l.side === 'debit').reduce((s, l) => s + l.amount, 0)
  const credit = lines.filter(l => l.side === 'credit').reduce((s, l) => s + l.amount, 0)
  return { valid: debit === credit && lines.length >= 2, debit, credit, diff: debit - credit }
}

import { useState, useEffect, useCallback } from 'react'
import { supabase } from '../supabase'
import type { EntryInput } from '../lib/journalEngine'

export interface JournalEntryWithLines {
  id: string
  tenant_id: string
  entry_date: string
  description: string
  status: 'draft' | 'posted' | 'void'
  source_type: string | null
  source_id: string | null
  created_at: string
  journal_lines: {
    id: string
    account_id: string
    side: 'debit' | 'credit'
    amount: number
    memo: string | null
    accounts: { code: string; name: string } | null
  }[]
}

interface Filters {
  status?: string
  dateFrom?: string
  dateTo?: string
}

export function useJournals(tenantId: string) {
  const [entries, setEntries] = useState<JournalEntryWithLines[]>([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

  const fetchEntries = useCallback(async (filters: Filters = {}) => {
    setLoading(true)
    setError(null)
    let q = supabase
      .from('journal_entries')
      .select(`*, journal_lines(*, accounts(code, name))`)
      .eq('tenant_id', tenantId)
      .order('entry_date', { ascending: false })
      .order('created_at', { ascending: false })
      .limit(200)

    if (filters.status)   q = q.eq('status', filters.status)
    if (filters.dateFrom) q = q.gte('entry_date', filters.dateFrom)
    if (filters.dateTo)   q = q.lte('entry_date', filters.dateTo)

    const { data, error: err } = await q
    if (err) setError(err.message)
    else setEntries((data ?? []) as JournalEntryWithLines[])
    setLoading(false)
  }, [tenantId])

  useEffect(() => { fetchEntries() }, [fetchEntries])

  // 전표 + 라인 동시 생성 (트랜잭션 아님 — Supabase 제약)
  const createEntry = async (input: EntryInput): Promise<string> => {
    const { data: entry, error: entryErr } = await supabase
      .from('journal_entries')
      .insert({
        tenant_id: input.tenant_id,
        entry_date: input.entry_date,
        description: input.description,
        status: 'draft',
        source_type: input.source_type ?? null,
        source_id: input.source_id ?? null,
      })
      .select()
      .single()

    if (entryErr || !entry) throw new Error(entryErr?.message ?? '전표 생성 실패')

    const { error: linesErr } = await supabase
      .from('journal_lines')
      .insert(input.lines.map(l => ({ ...l, journal_entry_id: entry.id })))

    if (linesErr) {
      // 라인 삽입 실패 시 헤더 삭제 (정합성 유지)
      await supabase.from('journal_entries').delete().eq('id', entry.id)
      throw new Error(linesErr.message)
    }

    await fetchEntries()
    return entry.id
  }

  // draft → posted (DB 트리거가 균형 검증)
  const postEntry = async (entryId: string): Promise<void> => {
    const { error: err } = await supabase
      .from('journal_entries')
      .update({ status: 'posted' })
      .eq('id', entryId)
      .eq('status', 'draft') // 이미 posted 이면 no-op

    if (err) throw new Error(err.message) // 트리거 EXCEPTION 이 여기로 전달됨
    await fetchEntries()
  }

  // posted → void (취소)
  const voidEntry = async (entryId: string): Promise<void> => {
    const { error: err } = await supabase
      .from('journal_entries')
      .update({ status: 'void' })
      .eq('id', entryId)
      .eq('status', 'posted')

    if (err) throw new Error(err.message)
    await fetchEntries()
  }

  // draft 삭제 (posted 는 삭제 불가 — void 사용)
  const deleteEntry = async (entryId: string): Promise<void> => {
    const { error: err } = await supabase
      .from('journal_entries')
      .delete()
      .eq('id', entryId)
      .eq('status', 'draft')

    if (err) throw new Error(err.message)
    await fetchEntries()
  }

  return { entries, loading, error, fetchEntries, createEntry, postEntry, voidEntry, deleteEntry }
}

import { useState } from 'react'
import { useJournals } from '../../hooks/useJournals'
import { validateBalance, EntryInput, LineInput } from '../../lib/journalEngine'
import { fmtKRW } from '../../constants/theme'

interface Account { id: string; code: string; name: string }
interface Props {
  tenantId: string
  accounts: Account[]
  onClose: () => void
  onSaved: () => void
}

const emptyLine = (): LineInput & { key: number } => ({
  key: Date.now() + Math.random(),
  account_id: '',
  side: 'debit',
  amount: 0,
})

export default function JournalFormModal({ tenantId, accounts, onClose, onSaved }: Props) {
  const { createEntry } = useJournals(tenantId)
  const [date, setDate] = useState(new Date().toISOString().slice(0, 10))
  const [description, setDescription] = useState('')
  const [lines, setLines] = useState([{ ...emptyLine(), side: 'debit' as const }, { ...emptyLine(), side: 'credit' as const }])
  const [saving, setSaving] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const updateLine = (idx: number, patch: Partial<LineInput>) => {
    setLines(ls => ls.map((l, i) => i === idx ? { ...l, ...patch } : l))
  }

  const balanceCheck = validateBalance(lines.map(l => ({ account_id: l.account_id, side: l.side, amount: l.amount })))

  const handleSubmit = async () => {
    setError(null)
    if (!description.trim()) { setError('설명을 입력하세요'); return }
    if (!balanceCheck.valid) {
      setError(`차대변 불일치: 차변 ${fmtKRW(balanceCheck.debit)} / 대변 ${fmtKRW(balanceCheck.credit)}`);
      return
    }

    setSaving(true)
    try {
      const input: EntryInput = {
        tenant_id: tenantId,
        entry_date: date,
        description,
        source_type: 'manual',
        lines: lines.map(l => ({ account_id: l.account_id, side: l.side, amount: l.amount, memo: undefined })),
      }
      await createEntry(input)
      onSaved()
      onClose()
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : '저장 실패')
    } finally {
      setSaving(false)
    }
  }

  return (
    <div className="fixed inset-0 bg-black/50 flex items-end md:items-center justify-center z-50">
      <div className="w-full md:max-w-lg bg-white dark:bg-gray-800 rounded-t-2xl md:rounded-2xl p-6">
        <div className="flex items-center justify-between mb-4">
          <h2 className="text-lg font-semibold dark:text-white">전표 입력</h2>
          <button onClick={onClose} className="text-gray-400 hover:text-gray-600 dark:hover:text-gray-200">✕</button>
        </div>

        <div className="space-y-3">
          <input type="date" value={date} onChange={e => setDate(e.target.value)}
            className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white" />
          <input type="text" value={description} onChange={e => setDescription(e.target.value)}
            placeholder="적요 (예: 임차료 6월분)"
            className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white" />

          <div className="space-y-2">
            {lines.map((line, idx) => (
              <div key={line.key} className="flex gap-2 items-center">
                <select value={line.side} onChange={e => updateLine(idx, { side: e.target.value as 'debit'|'credit' })}
                  className="w-20 px-2 py-1.5 border rounded dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm">
                  <option value="debit">차변</option>
                  <option value="credit">대변</option>
                </select>
                <select value={line.account_id} onChange={e => updateLine(idx, { account_id: e.target.value })}
                  className="flex-1 px-2 py-1.5 border rounded dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm">
                  <option value="">계정과목 선택</option>
                  {accounts.map(a => (
                    <option key={a.id} value={a.id}>{a.code} {a.name}</option>
                  ))}
                </select>
                <input type="number" value={line.amount || ''} onChange={e => updateLine(idx, { amount: parseInt(e.target.value) || 0 })}
                  placeholder="금액(원)"
                  className="w-28 px-2 py-1.5 border rounded dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm text-right" />
                {lines.length > 2 && (
                  <button onClick={() => setLines(ls => ls.filter((_, i) => i !== idx))}
                    className="text-red-400 hover:text-red-600 text-sm">✕</button>
                )}
              </div>
            ))}
          </div>

          <button onClick={() => setLines(ls => [...ls, emptyLine()])}
            className="text-sm text-blue-600 dark:text-blue-400 hover:underline">
            + 라인 추가
          </button>

          {/* 균형 표시 */}
          <div className={`text-sm ${balanceCheck.valid ? 'text-green-600' : 'text-red-500'}`}>
            차변 {fmtKRW(balanceCheck.debit)} / 대변 {fmtKRW(balanceCheck.credit)}
            {balanceCheck.valid ? ' ✓ 균형' : ` (차이: ${fmtKRW(Math.abs(balanceCheck.diff))})`}
          </div>

          {error && <p className="text-sm text-red-500">{error}</p>}

          <button onClick={handleSubmit} disabled={saving || !balanceCheck.valid}
            className="w-full py-2.5 bg-blue-600 hover:bg-blue-700 disabled:bg-blue-300
                       text-white font-medium rounded-lg transition-colors">
            {saving ? '저장 중...' : '임시저장'}
          </button>
        </div>
      </div>
    </div>
  )
}
