```typescript
// 5대 은행 CSV 파서 — 각 은행의 CSV 컬럼 매핑 구성

export interface ParsedTransaction {
  date: string         // YYYY-MM-DD
  description: string
  amount: number       // 원 단위, 음수=지출 / 양수=수입
  balance?: number
}

interface BankParser {
  name: string
  encoding: 'utf-8' | 'euc-kr'  // 실제로는 브라우저가 처리
  skipRows: number               // 헤더 행 수
  dateCol: number
  descCol: number
  withdrawalCol: number          // 출금 컬럼 (없으면 -1)
  depositCol: number             // 입금 컬럼 (없으면 -1)
  amountCol: number              // 단일 금액 컬럼 (부호 포함, -1이면 미사용)
  balanceCol: number             // 잔액 컬럼 (없으면 -1)
  dateFormat: 'YYYY.MM.DD' | 'YYYY-MM-DD' | 'YYYYMMDD' | 'YY.MM.DD'
}

export const BANK_PARSERS: Record<string, BankParser> = {
  kakaobank: {
    name: '카카오뱅크',
    encoding: 'utf-8',
    skipRows: 1,
    dateCol: 0, descCol: 2, withdrawalCol: 3, depositCol: 4, amountCol: -1, balanceCol: 5,
    dateFormat: 'YYYY.MM.DD',
  },
  shinhan: {
    name: '신한은행',
    encoding: 'euc-kr',
    skipRows: 1,
    dateCol: 0, descCol: 3, withdrawalCol: 4, depositCol: 5, amountCol: -1, balanceCol: 6,
    dateFormat: 'YYYY.MM.DD',
  },
  kookmin: {
    name: '국민은행',
    encoding: 'euc-kr',
    skipRows: 3,
    dateCol: 0, descCol: 2, withdrawalCol: 3, depositCol: 4, amountCol: -1, balanceCol: 5,
    dateFormat: 'YYYY-MM-DD',
  },
  woori: {
    name: '우리은행',
    encoding: 'euc-kr',
    skipRows: 1,
    dateCol: 0, descCol: 1, withdrawalCol: 2, depositCol: 3, amountCol: -1, balanceCol: 4,
    dateFormat: 'YYYY.MM.DD',
  },
  ibk: {
    name: '기업은행',
    encoding: 'euc-kr',
    skipRows: 2,
    dateCol: 0, descCol: 2, withdrawalCol: 3, depositCol: 4, amountCol: -1, balanceCol: 5,
    dateFormat: 'YYYYMMDD',
  },
}

function parseDate(raw: string, fmt: BankParser['dateFormat']): string {
  const clean = raw.trim()
  if (fmt === 'YYYY.MM.DD') return clean.replace(/\./g, '-')
  if (fmt === 'YYYY-MM-DD') return clean
  if (fmt === 'YYYYMMDD') return `${clean.slice(0,4)}-${clean.slice(4,6)}-${clean.slice(6,8)}`
  if (fmt === 'YY.MM.DD') return `20${clean.replace(/\./g, '-')}`
  return clean
}

function parseAmount(raw: string): number {
  return parseInt(raw.trim().replace(/[,\s]/g, ''), 10) || 0
}

export function parseBankCsv(csvText: string, bankKey: string): ParsedTransaction[] {
  const parser = BANK_PARSERS[bankKey]
  if (!parser) throw new Error(`지원하지 않는 은행: ${bankKey}`)

  const lines = csvText.split('\n').map(l => l.trim()).filter(Boolean)
  const rows = lines.slice(parser.skipRows)

  const results: ParsedTransaction[] = []

  for (const row of rows) {
    // CSV 파싱 (쌍따옴표 내 쉼표 처리)
    const cols: string[] = []
    let inQuote = false, cur = ''
    for (const ch of row) {
      if (ch === '"') { inQuote = !inQuote; continue }
      if (ch === ',' && !inQuote) { cols.push(cur); cur = ''; continue }
      cur += ch
    }
    cols.push(cur)

    if (cols.length < Math.max(parser.dateCol, parser.descCol) + 1) continue

    const date = parseDate(cols[parser.dateCol] ?? '', parser.dateFormat)
    const description = (cols[parser.descCol] ?? '').trim()
    if (!date || !description) continue

    let amount: number
    if (parser.amountCol >= 0) {
      amount = parseAmount(cols[parser.amountCol] ?? '0')
    } else {
      const withdrawal = parseAmount(cols[parser.withdrawalCol] ?? '0')
      const deposit    = parseAmount(cols[parser.depositCol]    ?? '0')
      amount = deposit > 0 ? deposit : -withdrawal
    }

    if (amount === 0) continue

    const balance = parser.balanceCol >= 0 ? parseAmount(cols[parser.balanceCol] ?? '0') : undefined
    results.push({ date, description, amount, balance })
  }

  return results
}
```

```typescript
// Bank Rules 자동 분류 엔진 — 순수 함수

export interface RuleCondition {
  field: 'description' | 'amount'
  op: 'contains' | 'not_contains' | 'eq' | 'gte' | 'lte'
  value: string | number
}

export interface BankRule {
  id: string
  name: string
  conditions: RuleCondition[]
  account_id: string
  description_template?: string
  priority: number
  is_active: boolean
}

export interface TxForRule {
  description: string
  amount: number
}

// 단일 트랜잭션에 규칙 조건이 모두 일치하는지 검사
function matchesCondition(tx: TxForRule, cond: RuleCondition): boolean {
  if (cond.field === 'description') {
    const desc = tx.description.toLowerCase()
    const val  = String(cond.value).toLowerCase()
    if (cond.op === 'contains')     return desc.includes(val)
    if (cond.op === 'not_contains') return !desc.includes(val)
    if (cond.op === 'eq')           return desc === val
  }
  if (cond.field === 'amount') {
    const amt = tx.amount
    const val = Number(cond.value)
    if (cond.op === 'eq')  return amt === val
    if (cond.op === 'gte') return amt >= val
    if (cond.op === 'lte') return amt <= val
  }
  return false
}

// 우선순위 순으로 정렬된 활성 규칙 중 첫 번째 매칭 규칙 반환
export function applyBankRules(tx: TxForRule, rules: BankRule[]): BankRule | null {
  const active = [...rules].filter(r => r.is_active).sort((a, b) => b.priority - a.priority)
  for (const rule of active) {
    if (rule.conditions.every(cond => matchesCondition(tx, cond))) {
      return rule
    }
  }
  return null
}

// 여러 트랜잭션에 규칙 일괄 적용
export function applyRulesToBatch(
  transactions: (TxForRule & { id: string })[],
  rules: BankRule[]
): { id: string; rule: BankRule | null }[] {
  return transactions.map(tx => ({ id: tx.id, rule: applyBankRules(tx, rules) }))
}
```

```typescript
import { useState, useRef } from 'react'
import { parseBankCsv, BANK_PARSERS } from '../../lib/bankCsvParser'
import { applyBankRules, BankRule } from '../../lib/bankRuleEngine'
import { supabase } from '../../supabase'
import { Card, Spinner } from '../../components/shared/CommonUI'
import { fmtKRW } from '../../constants/theme'

interface Props { tenantId: string }

export default function BankImportView({ tenantId }: Props) {
  const [selectedBank, setSelectedBank] = useState('kakaobank')
  const [preview, setPreview] = useState<{ date: string; description: string; amount: number }[]>([])
  const [importing, setImporting] = useState(false)
  const [importResult, setImportResult] = useState<{ total: number; matched: number } | null>(null)
  const [error, setError] = useState<string | null>(null)
  const fileRef = useRef<HTMLInputElement>(null)

  const handleFile = async (file: File) => {
    setError(null)
    setPreview([])
    setImportResult(null)

    const text = await file.text()
    try {
      const parsed = parseBankCsv(text, selectedBank)
      setPreview(parsed.slice(0, 5)) // 미리보기 5건
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : 'CSV 파싱 실패')
    }
  }

  const handleImport = async () => {
    if (!fileRef.current?.files?.[0]) return
    setImporting(true)
    setError(null)

    try {
      const file = fileRef.current.files[0]
      const text = await file.text()
      const parsed = parseBankCsv(text, selectedBank)

      // bank_imports 배치 헤더 생성
      const { data: importBatch, error: batchErr } = await supabase
        .from('bank_imports')
        .insert({
          tenant_id: tenantId,
          bank_name: BANK_PARSERS[selectedBank].name,
          filename: file.name,
          transaction_count: parsed.length,
        })
        .select()
        .single()

      if (batchErr || !importBatch) throw new Error(batchErr?.message ?? '배치 생성 실패')

      // Bank Rules 조회
      const { data: rules } = await supabase
        .from('bank_rules')
        .select('*')
        .eq('tenant_id', tenantId)
        .order('priority', { ascending: false })

      const activeRules: BankRule[] = (rules ?? []).map(r => ({
        ...r,
        conditions: r.conditions as typeof r.conditions,
      }))

      // bank_transactions 일괄 삽입
      const txRows = parsed.map(tx => {
        const matchedRule = applyBankRules(tx, activeRules)
        return {
          bank_import_id: importBatch.id,
          tenant_id: tenantId,
          transaction_date: tx.date,
          description: tx.description,
          amount: tx.amount,
          balance: tx.balance ?? null,
          status: matchedRule ? 'rule_applied' : 'unmatched',
          bank_rule_id: matchedRule?.id ?? null,
        }
      })

      // 500건 단위로 분할 삽입
      for (let i = 0; i < txRows.length; i += 500) {
        const { error: txErr } = await supabase.from('bank_transactions').insert(txRows.slice(i, i + 500))
        if (txErr) throw new Error(txErr.message)
      }

      const matchedCount = txRows.filter(r => r.status === 'rule_applied').length
      await supabase.from('bank_imports').update({ matched_count: matchedCount }).eq('id', importBatch.id)

      setImportResult({ total: parsed.length, matched: matchedCount })
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : '가져오기 실패')
    } finally {
      setImporting(false)
    }
  }

  return (
    <div className="p-4 max-w-2xl mx-auto">
      <h1 className="text-xl font-bold dark:text-white mb-4">은행 거래내역 업로드</h1>

      <Card className="p-4 space-y-4">
        <div>
          <label className="text-sm text-gray-600 dark:text-gray-400 block mb-1">은행 선택</label>
          <select value={selectedBank} onChange={e => setSelectedBank(e.target.value)}
            className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white">
            {Object.entries(BANK_PARSERS).map(([key, p]) => (
              <option key={key} value={key}>{p.name}</option>
            ))}
          </select>
        </div>

        <div>
          <label className="text-sm text-gray-600 dark:text-gray-400 block mb-1">CSV 파일</label>
          <input ref={fileRef} type="file" accept=".csv" onChange={e => e.target.files?.[0] && handleFile(e.target.files[0])}
            className="w-full text-sm text-gray-600 dark:text-gray-400
                       file:mr-3 file:py-1.5 file:px-3 file:rounded-lg file:border
                       file:text-sm file:font-medium file:cursor-pointer" />
        </div>

        {preview.length > 0 && (
          <div>
            <p className="text-xs text-gray-500 mb-2">미리보기 (첫 5건)</p>
            <div className="space-y-1">
              {preview.map((tx, i) => (
                <div key={i} className="flex justify-between text-xs text-gray-600 dark:text-gray-400 py-1 border-b dark:border-gray-700">
                  <span>{tx.date}</span>
                  <span className="flex-1 mx-2 truncate">{tx.description}</span>
                  <span className={tx.amount >= 0 ? 'text-blue-600' : 'text-gray-500'}>
                    {fmtKRW(tx.amount)}
                  </span>
                </div>
              ))}
            </div>
          </div>
        )}

        {error && <p className="text-sm text-red-500">{error}</p>}
        {importResult && (
          <p className="text-sm text-green-600">
            가져오기 완료: 총 {importResult.total}건 (규칙 자동매칭 {importResult.matched}건)
          </p>
        )}

        <button onClick={handleImport} disabled={importing || !preview.length}
          className="w-full py-2.5 bg-blue-600 hover:bg-blue-700 disabled:bg-blue-300
                     text-white font-medium rounded-lg transition-colors">
          {importing ? '가져오는 중...' : '가져오기'}
        </button>
      </Card>

      <p className="text-xs text-gray-400 mt-3 text-center">
        가져오기 후 <a href="/bank/reconcile" className="text-blue-500 hover:underline">대사 화면</a>에서 확인하세요
      </p>
    </div>
  )
}
```

```typescript
import { useState, useEffect } from 'react'
import { supabase } from '../../supabase'
import { Card, StatusBadge, EmptyState, Spinner } from '../../components/shared/CommonUI'
import { fmtKRW, fmtDate } from '../../constants/theme'

interface BankTx {
  id: string; transaction_date: string; description: string; amount: number
  status: string; bank_rules?: { name: string } | null
}

interface Props { tenantId: string }

type TabType = 'unmatched' | 'rule_applied' | 'manual'

export default function BankReconcileView({ tenantId }: Props) {
  const [tab, setTab] = useState<TabType>('unmatched')
  const [transactions, setTransactions] = useState<BankTx[]>([])
  const [loading, setLoading] = useState(true)
  const [updating, setUpdating] = useState<string | null>(null)

  const fetchTxs = async (status: string) => {
    setLoading(true)
    const { data } = await supabase
      .from('bank_transactions')
      .select('id, transaction_date, description, amount, status, bank_rules(name)')
      .eq('tenant_id', tenantId)
      .eq('status', status)
      .order('transaction_date', { ascending: false })
      .limit(100)
    setTransactions((data ?? []) as BankTx[])
    setLoading(false)
  }

  useEffect(() => { fetchTxs(tab) }, [tab, tenantId])

  const updateStatus = async (id: string, newStatus: string) => {
    setUpdating(id)
    await supabase.from('bank_transactions').update({ status: newStatus }).eq('id', id)
    setUpdating(null)
    fetchTxs(tab)
  }

  const TABS: { key: TabType; label: string; color: string }[] = [
    { key: 'unmatched',    label: '미처리',    color: 'text-red-500' },
    { key: 'rule_applied', label: '규칙 자동', color: 'text-blue-500' },
    { key: 'manual',       label: '수동 처리', color: 'text-purple-500' },
  ]

  return (
    <div className="p-4 max-w-3xl mx-auto">
      <h1 className="text-xl font-bold dark:text-white mb-4">은행 대사</h1>

      {/* 탭 */}
      <div className="flex gap-1 mb-4 bg-gray-100 dark:bg-gray-700 rounded-lg p-1">
        {TABS.map(t => (
          <button key={t.key} onClick={() => setTab(t.key)}
            className={`flex-1 py-1.5 rounded-md text-sm font-medium transition-colors ${
              tab === t.key
                ? 'bg-white dark:bg-gray-800 shadow-sm ' + t.color
                : 'text-gray-500 dark:text-gray-400'
            }`}>
            {t.label}
          </button>
        ))}
      </div>

      {loading ? <Spinner /> : (
        transactions.length === 0
          ? <EmptyState message={`${TABS.find(t => t.key === tab)?.label} 거래가 없습니다`} />
          : (
            <div className="space-y-2">
              {transactions.map(tx => (
                <Card key={tx.id} className="p-3">
                  <div className="flex items-center justify-between">
                    <div className="flex-1">
                      <div className="flex items-center gap-2 mb-0.5">
                        <span className="text-xs text-gray-400">{fmtDate(tx.transaction_date)}</span>
                        <StatusBadge status={tx.status} />
                        {tx.bank_rules && (
                          <span className="text-xs text-gray-400">규칙: {tx.bank_rules.name}</span>
                        )}
                      </div>
                      <p className="text-sm dark:text-white truncate">{tx.description}</p>
                      <p className={`text-sm font-semibold ${tx.amount >= 0 ? 'text-blue-600 dark:text-blue-400' : 'text-gray-500'}`}>
                        {fmtKRW(tx.amount)}
                      </p>
                    </div>
                    <div className="flex flex-col gap-1 ml-3">
                      {tab === 'unmatched' && (
                        <>
                          <button onClick={() => updateStatus(tx.id, 'manual')} disabled={!!updating}
                            className="text-xs px-2 py-1 bg-purple-50 dark:bg-purple-900/20 text-purple-700 dark:text-purple-400 border border-purple-200 dark:border-purple-800 rounded">
                            수동 처리
                          </button>
                          <button onClick={() => updateStatus(tx.id, 'ignored')} disabled={!!updating}
                            className="text-xs px-2 py-1 text-gray-400 border border-gray-200 dark:border-gray-700 rounded">
                            무시
                          </button>
                        </>
                      )}
                      {tab === 'rule_applied' && (
                        <button onClick={() => updateStatus(tx.id, 'manual')} disabled={!!updating}
                          className="text-xs px-2 py-1 text-gray-500 border border-gray-200 dark:border-gray-700 rounded">
                          수동 변경
                        </button>
                      )}
                      {updating === tx.id && (
                        <span className="text-xs text-gray-400">처리 중...</span>
                      )}
                    </div>
                  </div>
                </Card>
              ))}
            </div>
          )
      )}
    </div>
  )
}
```
