// 손익계산서 집계 엔진 — 순수 함수 (Supabase 없음)

export interface JournalLineForReport {
  amount: number
  side: 'debit' | 'credit'
  account_id: string
  accounts: { code: string; name: string; type: string } | null
  journal_entries: { entry_date: string; status: string } | null
}

export interface PLLine {
  accountCode: string
  accountName: string
  amount: number
}

export interface PLResult {
  period: string        // 'YYYY-MM' or 'YYYY-QN' or 'YYYY'
  revenue: PLLine[]     // 수입 계정별
  expenses: PLLine[]    // 지출 계정별
  totalRevenue: number
  totalExpenses: number
  netIncome: number
}

export interface MonthlyChartPoint {
  month: string         // 'MM월'
  수입: number
  지출: number
  순이익: number
}

// 라인 배열에서 월 필터링
function filterByMonth(lines: JournalLineForReport[], year: number, month: number): JournalLineForReport[] {
  const prefix = `${year}-${String(month).padStart(2, '0')}`
  return lines.filter(l => l.journal_entries?.entry_date.startsWith(prefix))
}

function filterByQuarter(lines: JournalLineForReport[], year: number, quarter: number): JournalLineForReport[] {
  const months = [1, 2, 3].map(m => m + (quarter - 1) * 3)
  return lines.filter(l => {
    const date = l.journal_entries?.entry_date ?? ''
    const m = parseInt(date.slice(5, 7), 10)
    return date.startsWith(String(year)) && months.includes(m)
  })
}

function filterByYear(lines: JournalLineForReport[], year: number): JournalLineForReport[] {
  return lines.filter(l => l.journal_entries?.entry_date.startsWith(String(year)))
}

// 계정 타입별 금액 집계
// 수입(revenue): credit 합계 - debit 합계 (순수입)
// 지출(expense): debit 합계 - credit 합계 (순지출)
function aggregateByAccount(
  lines: JournalLineForReport[],
  accountType: 'revenue' | 'expense'
): PLLine[] {
  const map = new Map<string, PLLine>()

  for (const line of lines) {
    const acct = line.accounts
    if (!acct || acct.type !== accountType) continue

    const key = acct.code
    if (!map.has(key)) {
      map.set(key, { accountCode: acct.code, accountName: acct.name, amount: 0 })
    }

    const entry = map.get(key)!
    if (accountType === 'revenue') {
      // 수입 계정: credit+, debit-
      entry.amount += line.side === 'credit' ? line.amount : -line.amount
    } else {
      // 지출 계정: debit+, credit-
      entry.amount += line.side === 'debit' ? line.amount : -line.amount
    }
  }

  return Array.from(map.values())
    .filter(l => l.amount !== 0)
    .sort((a, b) => a.accountCode.localeCompare(b.accountCode))
}

export function aggregatePL(
  lines: JournalLineForReport[],
  period: string,
  year: number,
  month?: number,
  quarter?: number
): PLResult {
  let filtered = lines
  if (month !== undefined) {
    filtered = filterByMonth(lines, year, month)
  } else if (quarter !== undefined) {
    filtered = filterByQuarter(lines, year, quarter)
  } else {
    filtered = filterByYear(lines, year)
  }

  const revenue  = aggregateByAccount(filtered, 'revenue')
  const expenses = aggregateByAccount(filtered, 'expense')
  const totalRevenue  = revenue.reduce((s, l) => s + l.amount, 0)
  const totalExpenses = expenses.reduce((s, l) => s + l.amount, 0)

  return {
    period,
    revenue,
    expenses,
    totalRevenue,
    totalExpenses,
    netIncome: totalRevenue - totalExpenses,
  }
}

// 최근 6개월 차트 데이터 빌드
export function buildMonthlyChart(lines: JournalLineForReport[]): MonthlyChartPoint[] {
  const points: MonthlyChartPoint[] = []
  const now = new Date()

  for (let i = 5; i >= 0; i--) {
    const d = new Date(now.getFullYear(), now.getMonth() - i, 1)
    const year  = d.getFullYear()
    const month = d.getMonth() + 1
    const period = `${year}-${String(month).padStart(2, '0')}`
    const pl = aggregatePL(lines, period, year, month)

    points.push({
      month: `${month}월`,
      수입: pl.totalRevenue,
      지출: pl.totalExpenses,
      순이익: pl.netIncome,
    })
  }

  return points
}

import { useState, useEffect } from 'react'
import { supabase } from '../../supabase'
import { Card, Spinner, EmptyState } from '../../components/shared/CommonUI'
import { fmtKRW, fmtDate } from '../../constants/theme'

interface Account { id: string; code: string; name: string; type: string }
interface LedgerLine {
  id: string; side: string; amount: number; memo: string | null
  journal_entries: { entry_date: string; description: string; status: string } | null
}

interface Props { tenantId: string }

export default function LedgerView({ tenantId }: Props) {
  const [accounts, setAccounts] = useState<Account[]>([])
  const [selectedId, setSelectedId] = useState('')
  const [lines, setLines] = useState<LedgerLine[]>([])
  const [loading, setLoading] = useState(false)

  useEffect(() => {
    supabase.from('accounts').select('id,code,name,type').eq('tenant_id', tenantId).order('code')
      .then(({ data }) => { setAccounts(data ?? []); if (data?.[0]) setSelectedId(data[0].id) })
  }, [tenantId])

  useEffect(() => {
    if (!selectedId) return
    setLoading(true)
    supabase.from('journal_lines')
      .select('id, side, amount, memo, journal_entries!inner(entry_date, description, status, tenant_id)')
      .eq('account_id', selectedId)
      .eq('journal_entries.tenant_id', tenantId)
      .eq('journal_entries.status', 'posted')
      .order('journal_entries(entry_date)', { ascending: false })
      .limit(100)
      .then(({ data }) => { setLines((data ?? []) as unknown as LedgerLine[]); setLoading(false) })
  }, [selectedId, tenantId])

  const selected = accounts.find(a => a.id === selectedId)
  const totalDebit  = lines.filter(l => l.side === 'debit').reduce((s, l) => s + l.amount, 0)
  const totalCredit = lines.filter(l => l.side === 'credit').reduce((s, l) => s + l.amount, 0)

  return (
    <div className="p-4 max-w-3xl mx-auto">
      <h1 className="text-xl font-bold dark:text-white mb-4">계정별 원장</h1>

      <div className="mb-4">
        <select value={selectedId} onChange={e => setSelectedId(e.target.value)}
          className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white">
          {accounts.map(a => (
            <option key={a.id} value={a.id}>{a.code} {a.name} ({a.type})</option>
          ))}
        </select>
      </div>

      {selected && lines.length > 0 && (
        <Card className="p-3 mb-4 grid grid-cols-2 gap-3">
          <div>
            <p className="text-xs text-gray-500">차변 합계</p>
            <p className="font-semibold text-gray-700 dark:text-white">{fmtKRW(totalDebit)}</p>
          </div>
          <div>
            <p className="text-xs text-gray-500">대변 합계</p>
            <p className="font-semibold text-gray-700 dark:text-white">{fmtKRW(totalCredit)}</p>
          </div>
        </Card>
      )}

      {loading ? <Spinner /> : lines.length === 0 ? (
        <EmptyState message="이 계정의 전표 라인이 없습니다 (posted 전표만 표시)" icon="📒" />
      ) : (
        <div className="space-y-1">
          {lines.map(line => {
            const entry = line.journal_entries
            return (
              <div key={line.id}
                className="flex items-center justify-between py-2 px-3 border-b dark:border-gray-700">
                <div className="flex-1">
                  <p className="text-xs text-gray-400">{entry ? fmtDate(entry.entry_date) : '-'}</p>
                  <p className="text-sm dark:text-white">{entry?.description}</p>
                  {line.memo && <p className="text-xs text-gray-400">{line.memo}</p>}
                </div>
                <div className="text-right ml-4">
                  {line.side === 'debit'
                    ? <p className="text-sm font-medium text-gray-700 dark:text-white">{fmtKRW(line.amount)}</p>
                    : <p className="text-xs text-gray-400">대변 {fmtKRW(line.amount)}</p>
                  }
                </div>
              </div>
            )
          })}
        </div>
      )}
    </div>
  )
}
