import { createClient } from '@supabase/supabase-js'

// ── Firebase JWT (서비스 계정) ────────────────────────────────
// Deno Web Crypto API 로 RSA-SHA256 JWT 서명
async function getFirebaseAccessToken(serviceAccount: {
  client_email: string
  private_key: string
}): Promise<string> {
  const now = Math.floor(Date.now() / 1000)
  const header = { alg: 'RS256', typ: 'JWT' }
  const payload = {
    iss: serviceAccount.client_email,
    sub: serviceAccount.client_email,
    aud: 'https://oauth2.googleapis.com/token',
    iat: now,
    exp: now + 3600,
    scope: 'https://www.googleapis.com/auth/datastore',
  }

  const encode = (obj: unknown) =>
    btoa(JSON.stringify(obj)).replace(/=/g, '').replace(/\+/g, '-').replace(/\//g, '_')

  const signingInput = `${encode(header)}.${encode(payload)}`

  const keyPem = serviceAccount.private_key
    .replace(/-----BEGIN PRIVATE KEY-----\n?/, '')
    .replace(/-----END PRIVATE KEY-----\n?/, '')
    .replace(/\n/g, '')

  const keyBytes = Uint8Array.from(atob(keyPem), c => c.charCodeAt(0))
  const cryptoKey = await crypto.subtle.importKey(
    'pkcs8', keyBytes,
    { name: 'RSASSA-PKCS1-v1_5', hash: 'SHA-256' },
    false, ['sign']
  )

  const sig = await crypto.subtle.sign(
    'RSASSA-PKCS1-v1_5', cryptoKey,
    new TextEncoder().encode(signingInput)
  )
  const sigB64 = btoa(String.fromCharCode(...new Uint8Array(sig)))
    .replace(/=/g, '').replace(/\+/g, '-').replace(/\//g, '_')

  const jwt = `${signingInput}.${sigB64}`

  // OAuth2 토큰 교환
  const tokenRes = await fetch('https://oauth2.googleapis.com/token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: `grant_type=urn%3Aietf%3Aparams%3Aoauth%3Agrant-type%3Ajwt-bearer&assertion=${jwt}`,
  })
  const tokenData = await tokenRes.json()
  if (!tokenData.access_token) {
    throw new Error(`Firebase OAuth 실패: ${JSON.stringify(tokenData)}`)
  }
  return tokenData.access_token
}

// ── Firestore 값 파싱 헬퍼 ────────────────────────────────────
function getStr(field: unknown): string {
  if (!field || typeof field !== 'object') return ''
  const f = field as Record<string, unknown>
  return (f.stringValue as string) ?? ''
}
function getNum(field: unknown): number {
  if (!field || typeof field !== 'object') return 0
  const f = field as Record<string, unknown>
  return parseInt((f.integerValue as string) ?? (f.doubleValue as string) ?? '0', 10)
}

// ── 메인 동기화 로직 ─────────────────────────────────────────
async function runSync(): Promise<{ newEntries: number; paymentCount: number }> {
  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  )

  const tenantId = Deno.env.get('RYEK_TENANT_ID')
  if (!tenantId) throw new Error('RYEK_TENANT_ID 환경변수 미설정')

  const serviceAccountJson = Deno.env.get('FIREBASE_SERVICE_ACCOUNT_JSON')
  if (!serviceAccountJson) throw new Error('FIREBASE_SERVICE_ACCOUNT_JSON 환경변수 미설정')
  const serviceAccount = JSON.parse(serviceAccountJson)

  // Firebase 접근 토큰 획득
  const accessToken = await getFirebaseAccessToken(serviceAccount)

  // Firestore rye-payments 조회
  const firestoreUrl = `https://firestore.googleapis.com/v1/projects/rye-k-center/databases/(default)/documents/appData/rye-payments`
  const fsRes = await fetch(firestoreUrl, {
    headers: { Authorization: `Bearer ${accessToken}` }
  })
  if (!fsRes.ok) throw new Error(`Firestore 조회 실패: ${fsRes.status}`)

  const fsData = await fsRes.json()
  const values = (fsData.fields?.value?.arrayValue?.values ?? []) as unknown[]

  // status='confirmed' 항목만 필터
  const confirmed = values
    .map((v: unknown) => (v as { mapValue?: { fields?: Record<string, unknown> } }).mapValue?.fields)
    .filter((f): f is Record<string, unknown> => !!f && getStr(f.status) === 'confirmed')

  // 계정과목 조회 (1020 보통예금, 4100 개인 수강료 수입)
  const { data: accts } = await supabase
    .from('accounts')
    .select('id, code')
    .eq('tenant_id', tenantId)
    .in('code', ['1020', '4100', '4200', '4300'])

  const acctMap = Object.fromEntries((accts ?? []).map(a => [a.code, a.id]))
  const debitAccountId  = acctMap['1020'] // 보통예금
  const creditAccountId = acctMap['4100'] // 개인 수강료 수입

  if (!debitAccountId || !creditAccountId) {
    throw new Error('필수 계정과목(1020, 4100) 미존재 — seed_default_accounts 실행 여부 확인')
  }

  let newEntries = 0

  for (const f of confirmed) {
    const paymentId   = getStr(f.id)
    const amount      = getNum(f.amount)
    const paidAtMs    = getNum(f.paidAt)
    const note        = getStr(f.note) || getStr(f.studentName) || paymentId
    const entryDate   = paidAtMs
      ? new Date(paidAtMs).toISOString().split('T')[0]
      : new Date().toISOString().split('T')[0]

    if (!paymentId || amount <= 0) continue

    // journal_entry 삽입 (멱등성: ON CONFLICT DO NOTHING)
    const { data: entry, error: entryErr } = await supabase
      .from('journal_entries')
      .insert({
        tenant_id: tenantId,
        entry_date: entryDate,
        description: `RYE-K 수납: ${note}`,
        status: 'draft',
        source_type: 'ryek_payment',
        source_id: paymentId,
      })
      .select('id')
      .single()

    // 중복 (23505) 또는 기타 에러 — 스킵
    if (entryErr?.code === '23505') continue
    if (entryErr || !entry) continue

    // journal_lines 삽입
    await supabase.from('journal_lines').insert([
      { journal_entry_id: entry.id, account_id: debitAccountId,  side: 'debit',  amount },
      { journal_entry_id: entry.id, account_id: creditAccountId, side: 'credit', amount },
    ])
    newEntries++
  }

  // 동기화 이력 기록
  await supabase.from('ryek_sync_log').insert({
    tenant_id: tenantId,
    payment_count: confirmed.length,
    new_entries: newEntries,
    status: 'success',
  })

  return { newEntries, paymentCount: confirmed.length }
}

// ── HTTP 핸들러 (수동 트리거용) ───────────────────────────────
Deno.serve(async (_req: Request) => {
  try {
    const result = await runSync()
    return new Response(JSON.stringify({ success: true, ...result }), {
      headers: { 'Content-Type': 'application/json' },
    })
  } catch (err: unknown) {
    const msg = err instanceof Error ? err.message : String(err)

    // 에러 로그 기록 시도
    try {
      const supabase = createClient(
        Deno.env.get('SUPABASE_URL')!,
        Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
      )
      const tenantId = Deno.env.get('RYEK_TENANT_ID')
      if (tenantId) {
        await supabase.from('ryek_sync_log').insert({
          tenant_id: tenantId, payment_count: 0, new_entries: 0,
          status: 'error', error_msg: msg,
        })
      }
    } catch { /* 로그 실패는 무시 */ }

    return new Response(JSON.stringify({ success: false, error: msg }), {
      status: 500, headers: { 'Content-Type': 'application/json' },
    })
  }
})

// ── 자동 스케줄 (5분마다) ────────────────────────────────────
// Deno.cron 은 Supabase Edge Runtime 에서 지원됨
Deno.cron('ryek-sync-5min', '*/5 * * * *', async () => {
  try {
    const result = await runSync()
    console.log(`[ryek-sync] 완료: ${result.newEntries}건 신규, ${result.paymentCount}건 처리`)
  } catch (err: unknown) {
    console.error('[ryek-sync] 에러:', err instanceof Error ? err.message : String(err))
  }
})
