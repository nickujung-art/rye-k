```typescript
import { useState, useEffect } from 'react'
import { User, Session } from '@supabase/supabase-js'
import { supabase } from '../supabase'

interface AuthState {
  user: User | null
  session: Session | null
  loading: boolean
}

export function useAuth() {
  const [state, setState] = useState<AuthState>({ user: null, session: null, loading: true })

  useEffect(() => {
    supabase.auth.getSession().then(({ data: { session } }) => {
      setState({ user: session?.user ?? null, session, loading: false })
    })

    const { data: { subscription } } = supabase.auth.onAuthStateChange((_event, session) => {
      setState({ user: session?.user ?? null, session, loading: false })
    })

    return () => subscription.unsubscribe()
  }, [])

  const signIn = async (email: string, password: string) => {
    const { error } = await supabase.auth.signInWithPassword({ email, password })
    if (error) throw new Error(error.message)
  }

  const signOut = async () => {
    await supabase.auth.signOut()
  }

  return { ...state, signIn, signOut }
}
```

**src/hooks/useTenant.ts:**
```typescript
import { useState, useEffect } from 'react'
import { User } from '@supabase/supabase-js'
import { supabase } from '../supabase'

interface TenantState {
  tenantId: string | null
  tenantName: string | null
  role: string | null
  loading: boolean
}

export function useTenant(user: User | null) {
  const [state, setState] = useState<TenantState>({
    tenantId: null, tenantName: null, role: null, loading: true
  })

  useEffect(() => {
    if (!user) {
      setState({ tenantId: null, tenantName: null, role: null, loading: false })
      return
    }
    loadOrCreateTenant(user)
  }, [user])

  async function loadOrCreateTenant(user: User) {
    // 1. 기존 테넌트 멤버십 조회
    const { data: membership } = await supabase
      .from('tenant_members')
      .select('tenant_id, role, tenants(id, name)')
      .eq('user_id', user.id)
      .single()

    if (membership) {
      const tenant = membership.tenants as { id: string; name: string } | null
      setState({
        tenantId: membership.tenant_id,
        tenantName: tenant?.name ?? null,
        role: membership.role,
        loading: false,
      })
      return
    }

    // 2. 첫 로그인 — 테넌트 자동 생성 (온보딩)
    const tenantName = user.email?.split('@')[0] ?? 'RYE-K 학원'
    const { data: newTenant, error: tenantErr } = await supabase
      .from('tenants')
      .insert({ name: tenantName })
      .select()
      .single()

    if (tenantErr || !newTenant) {
      console.error('테넌트 생성 실패:', tenantErr)
      setState({ tenantId: null, tenantName: null, role: null, loading: false })
      return
    }

    // 3. 멤버십 등록 (owner)
    await supabase.from('tenant_members').insert({
      tenant_id: newTenant.id,
      user_id: user.id,
      role: 'owner',
    })

    // 4. 26개 기본 계정과목 시드
    await supabase.rpc('seed_default_accounts', { p_tenant_id: newTenant.id })

    setState({
      tenantId: newTenant.id,
      tenantName: newTenant.name,
      role: 'owner',
      loading: false,
    })
  }

  return state
}
```

```typescript
function ProtectedApp() {
  const { user, loading: authLoading } = useAuth()
  const { tenantId, tenantName, loading: tenantLoading } = useTenant(user)

  if (authLoading || tenantLoading) return <Spinner />
  if (!user) return <Navigate to="/login" replace />

  return (
    <AppLayout tenantName={tenantName}>
      <Suspense fallback={<Spinner />}>
        <Routes>
          <Route path="/dashboard"      element={<DashboardView tenantId={tenantId!} />} />
          <Route path="/journal"        element={<JournalListView tenantId={tenantId!} />} />
          <Route path="/bank"           element={<BankImportView tenantId={tenantId!} />} />
          <Route path="/payroll"        element={<PayrollView tenantId={tenantId!} />} />
          <Route path="/invoices"       element={<InvoiceView tenantId={tenantId!} />} />
          <Route path="/reports/pl"     element={<ProfitLossView tenantId={tenantId!} />} />
          <Route path="/reports/ledger" element={<LedgerView tenantId={tenantId!} />} />
          <Route path="/reports/ar-aging" element={<ArAgingView tenantId={tenantId!} />} />
          <Route path="/recurring"      element={<RecurringView tenantId={tenantId!} />} />
          <Route path="*"               element={<Navigate to="/dashboard" replace />} />
        </Routes>
      </Suspense>
    </AppLayout>
  )
}

export default function App() {
  const { user, loading } = useAuth()

  if (loading) return <Spinner />

  return (
    <BrowserRouter>
      <Routes>
        <Route path="/login" element={user ? <Navigate to="/dashboard" replace /> : <LoginView />} />
        <Route path="/*"     element={<ProtectedApp />} />
      </Routes>
    </BrowserRouter>
  )
}
```
