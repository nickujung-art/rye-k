    **TimetableView.jsx 변경 4: handleCellAdd 함수 추가 + 빈 셀 렌더 수정**

    TimetableGrid 내부에 handleCellAdd 함수 추가 (props로 `selectedTeacherId`, `onAddStudentToSlot`, `students`, `teachers`를 받는 위치에서):

    ```js
    const [cellInstrument, setCellInstrument] = useState(null);

    const handleCellAdd = (day, time, e) => {
      if (!selectedTeacherId) {
        // 강사 미선택 — 빈 셀 "+" disabled 처리는 props 체크로 이미 제어
        // onAddStudentToSlot 없으면 버튼이 렌더되지 않으므로 이 분기는 fallback
        return;
      }
      const rect = e.currentTarget.getBoundingClientRect();
      // 해당 강사를 기준으로 이미 배정된 학생들의 악기 추출 (자동 선택)
      const existingInstruments = (students || [])
        .filter(s => (s.lessons || []).some(l => (l.teacherId || s.teacherId) === selectedTeacherId))
        .flatMap(s => (s.lessons || [])
          .filter(l => (l.teacherId || s.teacherId) === selectedTeacherId)
          .map(l => l.instrument)
        )
        .filter(Boolean);
      // 악기가 1종류면 자동 선택, 복수면 null(드롭다운)
      const uniqueInstruments = [...new Set(existingInstruments)];
      const autoInstrument = uniqueInstruments.length === 1 ? uniqueInstruments[0] : null;
      setCellInstrument(autoInstrument);
      setSearchPopup({ teacherId: selectedTeacherId, day, time, instrument: autoInstrument, top: rect.bottom + 6, left: rect.left });
    };
    ```
