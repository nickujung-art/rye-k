import { useState, useEffect } from 'react'
import { Link } from 'react-router-dom'
import { supabase } from '../supabase'
import { aggregatePL, buildMonthlyChart, JournalLineForReport } from '../lib/reportEngine'
import { Card, Spinner } from '../components/shared/CommonUI'
import { fmtKRW } from '../constants/theme'

interface Props { tenantId: string }

interface SyncStatus {
  lastSync: string | null
  paymentCount: number
  newEntries: number
  status: string
}

export default function DashboardView({ tenantId }: Props) {
  const [loading, setLoading] = useState(true)
  const [syncStatus, setSyncStatus] = useState<SyncStatus | null>(null)
  const [ryekAmount, setRyekAmount]   = useState(0)  // RYE-K posted 수납 총액 (이번달)
  const [ledgerAmount, setLedgerAmount] = useState(0) // 원장 수익 계정 total (이번달)
  const [chartData, setChartData] = useState<ReturnType<typeof buildMonthlyChart>>([])
  const [currentMonthPL, setCurrentMonthPL] = useState<{ revenue: number; expenses: number; net: number } | null>(null)

  const now = new Date()
  const year  = now.getFullYear()
  const month = now.getMonth() + 1
  const monthPrefix = `${year}-${String(month).padStart(2, '0')}`

  useEffect(() => {
    loadDashboard()
  }, [tenantId])

  const loadDashboard = async () => {
    setLoading(true)

    // 1. 최근 RYE-K 동기화 상태
    const { data: syncLog } = await supabase
      .from('ryek_sync_log')
      .select('synced_at, payment_count, new_entries, status')
      .eq('tenant_id', tenantId)
      .order('synced_at', { ascending: false })
      .limit(1)
      .single()

    if (syncLog) {
      setSyncStatus({ lastSync: syncLog.synced_at, paymentCount: syncLog.payment_count, newEntries: syncLog.new_entries, status: syncLog.status })
    }

    // 2. RYE-K 수납 금액 (이번 달 posted ryek_payment 전표 debit 합계)
    const { data: ryekLines } = await supabase
      .from('journal_lines')
      .select('amount, side, journal_entries!inner(entry_date, status, source_type, tenant_id)')
      .eq('journal_entries.tenant_id', tenantId)
      .eq('journal_entries.status', 'posted')
      .eq('journal_entries.source_type', 'ryek_payment')
      .gte('journal_entries.entry_date', `${monthPrefix}-01`)
      .lte('journal_entries.entry_date', `${monthPrefix}-31`)
      .eq('side', 'debit')

    const ryekTotal = (ryekLines ?? []).reduce((s, l) => s + l.amount, 0)
    setRyekAmount(ryekTotal)

    // 3. 원장 수익 계정 집계 (이번 달 posted, 4xxx revenue)
    const { data: allLines } = await supabase
      .from('journal_lines')
      .select('amount, side, account_id, accounts(code, name, type), journal_entries!inner(entry_date, status, tenant_id)')
      .eq('journal_entries.tenant_id', tenantId)
      .eq('journal_entries.status', 'posted')

    const typedLines = (allLines ?? []) as unknown as JournalLineForReport[]
    const pl = aggregatePL(typedLines, monthPrefix, year, month)
    setLedgerAmount(pl.totalRevenue)
    setCurrentMonthPL({ revenue: pl.totalRevenue, expenses: pl.totalExpenses, net: pl.netIncome })
    setChartData(buildMonthlyChart(typedLines))

    setLoading(false)
  }

  const diff = ryekAmount - ledgerAmount
  const hasMismatch = Math.abs(diff) > 0

  if (loading) return <Spinner />

  return (
    <div className="p-4 max-w-3xl mx-auto space-y-4">
      <div className="flex items-center justify-between">
        <h1 className="text-xl font-bold dark:text-white">대시보드</h1>
        <span className="text-xs text-gray-400">{monthPrefix}</span>
      </div>

      {/* 더블체크 섹션 — D-11 핵심 */}
      <Card className={`p-4 ${hasMismatch ? 'border-red-300 dark:border-red-700 bg-red-50 dark:bg-red-900/20' : 'border-green-200 dark:border-green-800 bg-green-50 dark:bg-green-900/20'}`}>
        <div className="flex items-center justify-between mb-3">
          <h2 className="text-sm font-semibold dark:text-white">더블체크</h2>
          {hasMismatch
            ? <span className="text-xs font-bold text-red-600 dark:text-red-400 bg-red-100 dark:bg-red-900/40 px-2 py-0.5 rounded-full">⚠ 차이 발생</span>
            : <span className="text-xs font-medium text-green-600 dark:text-green-400">✓ 일치</span>
          }
        </div>

        <div className="grid grid-cols-2 gap-3">
          <div>
            <p className="text-xs text-gray-500 dark:text-gray-400 mb-1">RYE-K 수납 (이번 달)</p>
            <p className="text-lg font-bold text-blue-600 dark:text-blue-400">{fmtKRW(ryekAmount)}</p>
            <p className="text-xs text-gray-400 mt-0.5">
              {syncStatus ? `${new Date(syncStatus.lastSync ?? '').toLocaleTimeString('ko-KR')} 동기화` : '미동기화'}
            </p>
          </div>
          <div>
            <p className="text-xs text-gray-500 dark:text-gray-400 mb-1">원장 수익 (이번 달)</p>
            <p className="text-lg font-bold text-blue-600 dark:text-blue-400">{fmtKRW(ledgerAmount)}</p>
            <Link to="/reports/pl" className="text-xs text-blue-500 hover:underline">원장 보기 →</Link>
          </div>
        </div>

        {hasMismatch && (
          <div className="mt-3 pt-3 border-t border-red-200 dark:border-red-800">
            <p className="text-sm text-red-700 dark:text-red-400 font-medium">
              차이: {fmtKRW(Math.abs(diff))} {diff > 0 ? '(원장 부족)' : '(원장 초과)'}
            </p>
            <p className="text-xs text-red-500 mt-1">
              RYE-K 동기화 후 draft 전표를 확인하세요.
              <Link to="/journal" className="ml-1 underline">전표 목록 →</Link>
            </p>
          </div>
        )}
      </Card>

      {/* 이번 달 손익 요약 */}
      {currentMonthPL && (
        <div className="grid grid-cols-3 gap-3">
          <Card className="p-3 text-center">
            <p className="text-xs text-gray-400">수입</p>
            <p className="text-base font-bold text-blue-600 dark:text-blue-400">{fmtKRW(currentMonthPL.revenue)}</p>
          </Card>
          <Card className="p-3 text-center">
            <p className="text-xs text-gray-400">지출</p>
            <p className="text-base font-bold text-gray-600 dark:text-gray-400">{fmtKRW(currentMonthPL.expenses)}</p>
          </Card>
          <Card className={`p-3 text-center ${currentMonthPL.net >= 0 ? 'bg-green-50 dark:bg-green-900/20' : 'bg-red-50 dark:bg-red-900/20'}`}>
            <p className="text-xs text-gray-400">순이익</p>
            <p className={`text-base font-bold ${currentMonthPL.net >= 0 ? 'text-green-600 dark:text-green-400' : 'text-red-600 dark:text-red-400'}`}>
              {fmtKRW(currentMonthPL.net)}
            </p>
          </Card>
        </div>
      )}

      {/* 빠른 링크 */}
      <div className="grid grid-cols-2 gap-3">
        <Link to="/journal" className="block">
          <Card className="p-3 hover:bg-gray-50 dark:hover:bg-gray-700 transition-colors">
            <p className="text-sm font-medium dark:text-white">✏ 전표 입력</p>
            <p className="text-xs text-gray-400 mt-0.5">수동 분개 + 임시저장</p>

import { createClient } from '@supabase/supabase-js'

interface TaxEmailBody {
  tenantId: string
  year: number
  month: number
  recipientEmail: string
  plSummary: {
    revenue: number
    expenses: number
    netIncome: number
    revenueLines: { code: string; name: string; amount: number }[]
    expenseLines: { code: string; name: string; amount: number }[]
  }
  withholdingSummary: {
    totalWithholding: number
    teacherCount: number
  }
}

function buildEmailHtml(body: TaxEmailBody): string {
  const { year, month, plSummary, withholdingSummary } = body
  const fmt = (n: number) => new Intl.NumberFormat('ko-KR', { style: 'currency', currency: 'KRW' }).format(n)

  const revenueRows = plSummary.revenueLines
    .map(l => `<tr><td style="padding:4px 8px">${l.code} ${l.name}</td><td style="padding:4px 8px;text-align:right">${fmt(l.amount)}</td></tr>`)
    .join('')

  const expenseRows = plSummary.expenseLines
    .map(l => `<tr><td style="padding:4px 8px">${l.code} ${l.name}</td><td style="padding:4px 8px;text-align:right">${fmt(l.amount)}</td></tr>`)
    .join('')

  return `
<!DOCTYPE html>
<html lang="ko">
<head><meta charset="UTF-8"><title>RYE-K 월별 세무 자료</title></head>
<body style="font-family:sans-serif;max-width:600px;margin:0 auto;padding:20px;color:#111">
  <h2 style="color:#1d4ed8">RYE-K 국악 학원 — ${year}년 ${month}월 세무 자료</h2>
  <p style="color:#6b7280;font-size:14px">자동 생성 자료입니다. 검토 후 신고에 활용해 주세요.</p>

  <h3 style="border-bottom:2px solid #e5e7eb;padding-bottom:8px">손익계산서</h3>
  <table style="width:100%;border-collapse:collapse;font-size:14px">
    <tr style="background:#f9fafb"><th style="padding:4px 8px;text-align:left">항목</th><th style="padding:4px 8px;text-align:right">금액</th></tr>
    <tr><td colspan="2" style="padding:4px 8px;font-weight:bold;background:#eff6ff;color:#1d4ed8">수입</td></tr>
    ${revenueRows}
    <tr style="font-weight:bold;border-top:1px solid #e5e7eb">
      <td style="padding:4px 8px">수입 합계</td>
      <td style="padding:4px 8px;text-align:right;color:#1d4ed8">${fmt(plSummary.revenue)}</td>
    </tr>
    <tr><td colspan="2" style="padding:4px 8px;font-weight:bold;background:#f9fafb">지출</td></tr>
    ${expenseRows}
    <tr style="font-weight:bold;border-top:1px solid #e5e7eb">
      <td style="padding:4px 8px">지출 합계</td>
      <td style="padding:4px 8px;text-align:right">${fmt(plSummary.expenses)}</td>
    </tr>
    <tr style="font-weight:bold;font-size:16px;border-top:2px solid #111">
      <td style="padding:8px">순이익</td>
      <td style="padding:8px;text-align:right;color:${plSummary.netIncome >= 0 ? '#16a34a' : '#dc2626'}">
        ${fmt(plSummary.netIncome)}
      </td>
    </tr>
  </table>

  <h3 style="border-bottom:2px solid #e5e7eb;padding-bottom:8px;margin-top:24px">원천세 집계</h3>
  <table style="width:100%;border-collapse:collapse;font-size:14px">
    <tr><td style="padding:4px 8px">사업소득 원천세 대상 인원</td><td style="padding:4px 8px;text-align:right">${withholdingSummary.teacherCount}명</td></tr>
    <tr style="font-weight:bold"><td style="padding:4px 8px">이번 달 원천세 합계 (익월 10일 신고)</td><td style="padding:4px 8px;text-align:right;color:#dc2626">${fmt(withholdingSummary.totalWithholding)}</td></tr>
  </table>

  <p style="font-size:12px;color:#9ca3af;margin-top:24px">
    이 이메일은 RYE-K 회계 시스템에서 자동 발송되었습니다.<br>
    회계 시스템: accounting.ryekorea.com
  </p>

Deno.serve(async (req: Request) => {
  const body: TaxEmailBody = await req.json()

  const resendApiKey = Deno.env.get('RESEND_API_KEY')
  const fromEmail    = Deno.env.get('FROM_EMAIL') ?? 'accounting@ryekorea.com'

  if (!resendApiKey) {
    return new Response(JSON.stringify({ success: false, error: 'RESEND_API_KEY 미설정' }), {
      status: 500, headers: { 'Content-Type': 'application/json' }
    })
  }

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  )

  // 이메일 발송 (Resend API)
  const emailRes = await fetch('https://api.resend.com/emails', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${resendApiKey}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      from: fromEmail,
      to: [body.recipientEmail],
      subject: `[RYE-K] ${body.year}년 ${body.month}월 세무 자료`,
      html: buildEmailHtml(body),
    }),
  })

  const emailData = await emailRes.json()
  const success = emailRes.ok

  // tax_exports 이력 기록
  await supabase.from('tax_exports').insert({
    tenant_id: body.tenantId,
    period_year: body.year,
    period_month: body.month,
    sent_at: success ? new Date().toISOString() : null,
    recipient_email: body.recipientEmail,
    status: success ? 'sent' : 'error',
  })

  return new Response(
    JSON.stringify({ success, emailId: emailData.id, error: emailData.message }),
    { headers: { 'Content-Type': 'application/json' } }
  )
})

import { useState, useEffect } from 'react'
import { supabase } from '../../supabase'
import { aggregatePL, JournalLineForReport } from '../../lib/reportEngine'
import { Card, Spinner, EmptyState } from '../../components/shared/CommonUI'
import { fmtKRW } from '../../constants/theme'

interface TaxExport { id: string; period_year: number; period_month: number; sent_at: string|null; recipient_email: string|null; status: string }
interface Props { tenantId: string }

export default function TaxExportView({ tenantId }: Props) {
  const [exports, setExports] = useState<TaxExport[]>([])
  const [loading, setLoading] = useState(true)
  const [sending, setSending] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const [successMsg, setSuccessMsg] = useState<string | null>(null)

  // 전송 설정
  const now = new Date()
  const [selectedYear, setSelectedYear]   = useState(now.getFullYear())
  const [selectedMonth, setSelectedMonth] = useState(now.getMonth() + 1)
  const [recipientEmail, setRecipientEmail] = useState('')

  const fetchExports = async () => {
    const { data } = await supabase
      .from('tax_exports').select('*')
      .eq('tenant_id', tenantId)
      .order('created_at', { ascending: false })
      .limit(24)
    setExports((data ?? []) as TaxExport[])
    setLoading(false)
  }

  // 테넌트 세무사 이메일 자동 로드
  useEffect(() => {
    fetchExports()
    supabase.from('tenants').select('tax_accountant_email').eq('id', tenantId).single()
      .then(({ data }) => { if (data?.tax_accountant_email) setRecipientEmail(data.tax_accountant_email) })
  }, [tenantId])

  const handleSend = async () => {
    if (!recipientEmail.trim()) { setError('수신 이메일을 입력하세요'); return }
    setSending(true)
    setError(null)

    try {
      // 해당 월 P&L 데이터 조회
      const { data: allLines } = await supabase
        .from('journal_lines')
        .select('amount, side, account_id, accounts(code, name, type), journal_entries!inner(entry_date, status, tenant_id)')
        .eq('journal_entries.tenant_id', tenantId)
        .eq('journal_entries.status', 'posted')

      const typedLines = (allLines ?? []) as unknown as JournalLineForReport[]
      const pl = aggregatePL(typedLines, `${selectedYear}-${selectedMonth}`, selectedYear, selectedMonth)

      // 이번 달 원천세 집계
      const monthPrefix = `${selectedYear}-${String(selectedMonth).padStart(2, '0')}`
      const { data: payrollData } = await supabase
        .from('payroll_records')
        .select('withholding_tax')
        .eq('tenant_id', tenantId)
        .gte('pay_date', `${monthPrefix}-01`)
        .lte('pay_date', `${monthPrefix}-31`)

      const totalWithholding = (payrollData ?? []).reduce((s, r) => s + r.withholding_tax, 0)

      // Edge Function 호출
      const { data: result, error: fnErr } = await supabase.functions.invoke('tax-email', {
        body: {
          tenantId,
          year: selectedYear,
          month: selectedMonth,
          recipientEmail: recipientEmail.trim(),
          plSummary: {
            revenue: pl.totalRevenue,
            expenses: pl.totalExpenses,
            netIncome: pl.netIncome,
            revenueLines: pl.revenue,
            expenseLines: pl.expenses,
          },
          withholdingSummary: {
            totalWithholding,
            teacherCount: payrollData?.length ?? 0,
          },
        },
      })

      if (fnErr || !result?.success) {
        throw new Error(result?.error ?? fnErr?.message ?? '전송 실패')
      }

      setSuccessMsg(`${selectedYear}년 ${selectedMonth}월 세무 자료를 ${recipientEmail} 로 전송했습니다`)
      setTimeout(() => setSuccessMsg(null), 5000)
      fetchExports()
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : '전송 실패')
    } finally {
      setSending(false)
    }
  }

  const currentYear = new Date().getFullYear()

  return (
    <div className="p-4 max-w-2xl mx-auto space-y-4">
      <h1 className="text-xl font-bold dark:text-white">세무사 이메일 전송</h1>

      <Card className="p-4 space-y-3">
        <h2 className="text-sm font-semibold dark:text-white">전송 설정</h2>

        <div className="flex gap-2">
          <select value={selectedYear} onChange={e => setSelectedYear(parseInt(e.target.value))}
            className="px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm">
            {[0,1,2].map(i => <option key={i} value={currentYear-i}>{currentYear-i}년</option>)}
          </select>
          <select value={selectedMonth} onChange={e => setSelectedMonth(parseInt(e.target.value))}
            className="px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm">
            {Array.from({length:12},(_,i)=>i+1).map(m=><option key={m} value={m}>{m}월</option>)}
          </select>
        </div>

        <input type="email" value={recipientEmail} onChange={e => setRecipientEmail(e.target.value)}
          placeholder="세무사 이메일 주소"
          className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm" />

        {error && <p className="text-sm text-red-500">{error}</p>}
        {successMsg && <p className="text-sm text-green-600">{successMsg}</p>}

        <button onClick={handleSend} disabled={sending}
          className="w-full py-2.5 bg-blue-600 hover:bg-blue-700 disabled:bg-blue-300
                     text-white font-medium rounded-lg transition-colors">
          {sending ? '전송 중...' : `${selectedYear}년 ${selectedMonth}월 세무 자료 전송`}
        </button>
      </Card>

      {/* 전송 이력 */}
      <div>
        <h2 className="text-sm font-semibold dark:text-white mb-2">전송 이력</h2>
        {loading ? <Spinner /> : exports.length === 0 ? (
          <EmptyState message="전송 이력이 없습니다" icon="📧" />
        ) : (
          <div className="space-y-2">
            {exports.map(ex => (
              <Card key={ex.id} className="p-3">
                <div className="flex justify-between items-center">
                  <div>
                    <p className="text-sm font-medium dark:text-white">{ex.period_year}년 {ex.period_month}월</p>
                    <p className="text-xs text-gray-400">{ex.recipient_email ?? '-'}</p>
                  </div>
                  <span className={`text-xs px-2 py-0.5 rounded-full font-medium ${
                    ex.status === 'sent' ? 'bg-green-100 dark:bg-green-900/30 text-green-700 dark:text-green-400' :
                    ex.status === 'error' ? 'bg-red-100 dark:bg-red-900/30 text-red-700 dark:text-red-400' :
                    'bg-gray-100 dark:bg-gray-700 text-gray-600 dark:text-gray-400'
                  }`}>
                    {ex.status === 'sent' ? '전송완료' : ex.status === 'error' ? '실패' : '대기'}
                  </span>
                </div>
              </Card>
            ))}
          </div>
        )}
      </div>
    </div>
  )
}
