-- ============================================================
-- 복식부기 균형 검증 트리거
-- draft→posted 전환 시 차변합계 ≠ 대변합계이면 EXCEPTION
-- ============================================================
CREATE OR REPLACE FUNCTION validate_journal_balance()
RETURNS TRIGGER AS $$
DECLARE
  v_debit  BIGINT;
  v_credit BIGINT;
BEGIN
  IF NEW.status = 'posted' AND (OLD.status IS NULL OR OLD.status = 'draft') THEN
    SELECT
      COALESCE(SUM(CASE WHEN side = 'debit'  THEN amount ELSE 0 END), 0),
      COALESCE(SUM(CASE WHEN side = 'credit' THEN amount ELSE 0 END), 0)
    INTO v_debit, v_credit
    FROM journal_lines
    WHERE journal_entry_id = NEW.id;

    IF v_debit = 0 AND v_credit = 0 THEN
      RAISE EXCEPTION '전표 %에 라인이 없습니다. 최소 1개의 라인이 필요합니다.', NEW.id;
    END IF;
    IF v_debit != v_credit THEN
      RAISE EXCEPTION '복식부기 불균형: 차변=% 대변=% (전표 %)', v_debit, v_credit, NEW.id;
    END IF;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_journal_balance
  BEFORE UPDATE ON journal_entries
  FOR EACH ROW EXECUTE FUNCTION validate_journal_balance();

-- updated_at 자동 갱신
CREATE OR REPLACE FUNCTION fn_update_updated_at()
RETURNS TRIGGER AS $$
BEGIN NEW.updated_at = NOW(); RETURN NEW; END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_je_updated_at

RETURNS UUID AS $$
  SELECT tenant_id FROM tenant_members WHERE user_id = auth.uid() LIMIT 1;
$$ LANGUAGE sql SECURITY DEFINER STABLE;
