```typescript
import { useState, FormEvent } from 'react'
import { createExpenseEntry } from '../../lib/journalEngine'
import { useJournals } from '../../hooks/useJournals'
import { fmtKRW } from '../../constants/theme'

interface Account { id: string; code: string; name: string }

interface Props {
  tenantId: string
  expenseAccounts: Account[]  // 5xxx 계정과목만 필터링
  bankAccountId: string        // 1020 보통예금 id
  onClose: () => void
  onSaved: () => void
}

export default function QuickExpenseView({ tenantId, expenseAccounts, bankAccountId, onClose, onSaved }: Props) {
  const { createEntry } = useJournals(tenantId)
  const [date, setDate]        = useState(new Date().toISOString().slice(0, 10))
  const [accountId, setAccountId] = useState(expenseAccounts[0]?.id ?? '')
  const [amount, setAmount]    = useState('')
  const [memo, setMemo]        = useState('')
  const [saving, setSaving]    = useState(false)
  const [error, setError]      = useState<string | null>(null)

  const parsedAmount = parseInt(amount.replace(/,/g, ''), 10) || 0

  const handleSubmit = async (e: FormEvent) => {
    e.preventDefault()
    setError(null)

    if (!accountId)       { setError('계정과목을 선택하세요'); return }
    if (parsedAmount <= 0) { setError('금액을 입력하세요'); return }
    if (!bankAccountId)   { setError('보통예금(1020) 계정이 없습니다. 계정과목 설정을 확인하세요'); return }

    const selectedAccount = expenseAccounts.find(a => a.id === accountId)
    const description = memo.trim() || selectedAccount?.name || '지출'

    setSaving(true)
    try {
      const entry = createExpenseEntry({
        tenant_id: tenantId,
        expense_account_id: accountId,
        bank_account_id: bankAccountId,
        amount: parsedAmount,
        description,
        entry_date: date,
      })
      await createEntry(entry)
      onSaved()
      onClose()
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : '저장 실패')
    } finally {
      setSaving(false)
    }
  }

  return (
    <div className="fixed inset-0 bg-black/50 flex items-end md:items-center justify-center z-50">
      <div className="w-full md:max-w-sm bg-white dark:bg-gray-800 rounded-t-2xl md:rounded-2xl p-6">
        <div className="flex items-center justify-between mb-4">
          <h2 className="text-lg font-semibold dark:text-white">지출 입력</h2>
          <button onClick={onClose} className="text-gray-400 hover:text-gray-600 dark:hover:text-gray-200 text-xl">✕</button>
        </div>

        <form onSubmit={handleSubmit} className="space-y-3">
          <div>
            <label className="text-sm text-gray-600 dark:text-gray-400 block mb-1">날짜</label>
            <input type="date" value={date} onChange={e => setDate(e.target.value)}
              className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white" />
          </div>

          <div>
            <label className="text-sm text-gray-600 dark:text-gray-400 block mb-1">계정과목</label>
            <select value={accountId} onChange={e => setAccountId(e.target.value)}
              className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white">
              {expenseAccounts.map(a => (
                <option key={a.id} value={a.id}>{a.code} {a.name}</option>
              ))}
            </select>
          </div>

          <div>
            <label className="text-sm text-gray-600 dark:text-gray-400 block mb-1">금액</label>
            <input
              type="text"
              inputMode="numeric"
              value={amount}
              onChange={e => {
                const raw = e.target.value.replace(/[^\d]/g, '')
                setAmount(raw ? parseInt(raw, 10).toLocaleString('ko-KR') : '')
              }}
              placeholder="0"
              className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-right"
            />
            {parsedAmount > 0 && (
              <p className="text-xs text-gray-400 text-right mt-0.5">{fmtKRW(parsedAmount)}</p>
            )}
          </div>

          <div>
            <label className="text-sm text-gray-600 dark:text-gray-400 block mb-1">메모 (선택)</label>
            <input type="text" value={memo} onChange={e => setMemo(e.target.value)}
              placeholder="예: 6월 임차료"
              className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white" />
          </div>

          {error && <p className="text-sm text-red-500">{error}</p>}

          <button type="submit" disabled={saving}
            className="w-full py-3 bg-blue-600 hover:bg-blue-700 disabled:bg-blue-300
                       text-white font-medium rounded-xl transition-colors text-base">
            {saving ? '저장 중...' : '저장'}
          </button>
        </form>
      </div>
    </div>
  )
```

**src/hooks/useRecurring.ts:**
```typescript
import { useState, useEffect, useCallback } from 'react'
import { supabase } from '../supabase'
import { createExpenseEntry } from '../lib/journalEngine'
import { useJournals } from './useJournals'

export interface RecurringTemplate {
  id: string
  tenant_id: string
  name: string
  day_of_month: number
  account_id: string
  amount: number
  description: string | null
  is_active: boolean
  accounts?: { code: string; name: string } | null
}

export function useRecurring(tenantId: string) {
  const [templates, setTemplates] = useState<RecurringTemplate[]>([])
  const [loading, setLoading] = useState(true)
  const { createEntry } = useJournals(tenantId)

  const fetchTemplates = useCallback(async () => {
    const { data } = await supabase
      .from('recurring_templates')
      .select('*, accounts(code, name)')
      .eq('tenant_id', tenantId)
      .order('day_of_month')
    setTemplates((data ?? []) as RecurringTemplate[])
    setLoading(false)
  }, [tenantId])

  useEffect(() => { fetchTemplates() }, [fetchTemplates])

  const addTemplate = async (input: Omit<RecurringTemplate, 'id' | 'tenant_id' | 'accounts'>) => {
    const { error } = await supabase.from('recurring_templates').insert({ ...input, tenant_id: tenantId })
    if (error) throw new Error(error.message)
    await fetchTemplates()
  }

  const toggleActive = async (id: string, isActive: boolean) => {
    const { error } = await supabase.from('recurring_templates').update({ is_active: isActive }).eq('id', id)
    if (error) throw new Error(error.message)
    await fetchTemplates()
  }

  const deleteTemplate = async (id: string) => {
    const { error } = await supabase.from('recurring_templates').delete().eq('id', id)
    if (error) throw new Error(error.message)
    await fetchTemplates()
  }

  // 이번 달 초안 생성 (bank_account_id 는 호출자가 제공)
  const generateDraftForMonth = async (template: RecurringTemplate, bankAccountId: string) => {
    const now = new Date()
    const year = now.getFullYear()
    const month = now.getMonth() + 1
    const day = Math.min(template.day_of_month, new Date(year, month, 0).getDate())
    const entryDate = `${year}-${String(month).padStart(2, '0')}-${String(day).padStart(2, '0')}`

    const entry = createExpenseEntry({
      tenant_id: tenantId,
      expense_account_id: template.account_id,
      bank_account_id: bankAccountId,
      amount: template.amount,
      description: template.description ?? template.name,
      entry_date: entryDate,
    })

    return createEntry(entry)
  }

  return { templates, loading, fetchTemplates, addTemplate, toggleActive, deleteTemplate, generateDraftForMonth }
}
```

```typescript
import { useState, useEffect } from 'react'
import { useRecurring, RecurringTemplate } from '../../hooks/useRecurring'
import { Card, Spinner, EmptyState } from '../../components/shared/CommonUI'
import { fmtKRW } from '../../constants/theme'
import { supabase } from '../../supabase'

interface Props { tenantId: string }

export default function RecurringView({ tenantId }: Props) {
  const { templates, loading, addTemplate, toggleActive, deleteTemplate, generateDraftForMonth } = useRecurring(tenantId)
  const [accounts, setAccounts] = useState<{ id: string; code: string; name: string }[]>([])
  const [bankAccountId, setBankAccountId] = useState('')
  const [showAdd, setShowAdd] = useState(false)
  const [saving, setSaving] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const [successMsg, setSuccessMsg] = useState<string | null>(null)

  // 신규 템플릿 폼 상태
  const [newName, setNewName] = useState('')
  const [newDay, setNewDay] = useState(1)
  const [newAccountId, setNewAccountId] = useState('')
  const [newAmount, setNewAmount] = useState('')

  useEffect(() => {
    supabase.from('accounts').select('id,code,name').eq('tenant_id', tenantId).order('code').then(({ data }) => {
      setAccounts(data ?? [])
      const bank = data?.find(a => a.code === '1020')
      if (bank) setBankAccountId(bank.id)
    })
  }, [tenantId])

  const expenseAccounts = accounts.filter(a => a.code.startsWith('5'))

  const handleAdd = async () => {
    if (!newName || !newAccountId || !newAmount) { setError('모든 필드를 입력하세요'); return }
    setSaving(true)
    setError(null)
    try {
      await addTemplate({
        name: newName,
        day_of_month: newDay,
        account_id: newAccountId,
        amount: parseInt(newAmount.replace(/,/g, ''), 10),
        description: newName,
        is_active: true,
      })
      setShowAdd(false)
      setNewName(''); setNewDay(1); setNewAccountId(''); setNewAmount('')
    } catch (e: unknown) { setError(e instanceof Error ? e.message : '추가 실패') }
    finally { setSaving(false) }
  }

  const handleGenerate = async (template: RecurringTemplate) => {
    if (!bankAccountId) { setError('보통예금(1020) 계정을 찾을 수 없습니다'); return }
    setSaving(true)
    setError(null)
    try {
      await generateDraftForMonth(template, bankAccountId)
      setSuccessMsg(`'${template.name}' 이번 달 초안이 생성되었습니다`)
      setTimeout(() => setSuccessMsg(null), 3000)
    } catch (e: unknown) { setError(e instanceof Error ? e.message : '초안 생성 실패') }
    finally { setSaving(false) }
  }

  if (loading) return <Spinner />

  return (
    <div className="p-4 max-w-2xl mx-auto">
      <div className="flex items-center justify-between mb-4">
        <h1 className="text-xl font-bold dark:text-white">반복 전표</h1>
        <button onClick={() => setShowAdd(s => !s)}
          className="px-3 py-1.5 bg-blue-600 text-white text-sm rounded-lg hover:bg-blue-700">
          + 추가
        </button>
      </div>

      {error && <p className="text-sm text-red-500 mb-3">{error}</p>}
      {successMsg && <p className="text-sm text-green-600 mb-3">{successMsg}</p>}

      {showAdd && (
        <Card className="p-4 mb-4 space-y-3">
          <h2 className="text-sm font-semibold dark:text-white">새 반복 템플릿</h2>
          <input type="text" value={newName} onChange={e => setNewName(e.target.value)}
            placeholder="이름 (예: 월 임차료)"
            className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm" />
          <div className="flex gap-2">
            <div className="flex-1">
              <label className="text-xs text-gray-500 block mb-1">매월</label>
              <select value={newDay} onChange={e => setNewDay(parseInt(e.target.value))}
                className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm">
                {Array.from({ length: 28 }, (_, i) => i + 1).map(d => (
                  <option key={d} value={d}>{d}일</option>
                ))}
              </select>
            </div>
            <div className="flex-1">
              <label className="text-xs text-gray-500 block mb-1">계정과목</label>
              <select value={newAccountId} onChange={e => setNewAccountId(e.target.value)}
                className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm">
                <option value="">선택</option>
                {expenseAccounts.map(a => <option key={a.id} value={a.id}>{a.code} {a.name}</option>)}
              </select>
            </div>
          </div>
          <input type="text" inputMode="numeric" value={newAmount}
            onChange={e => { const raw = e.target.value.replace(/\D/g, ''); setNewAmount(raw ? parseInt(raw).toLocaleString('ko-KR') : '') }}
            placeholder="금액"
            className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm text-right" />
          <button onClick={handleAdd} disabled={saving}
            className="w-full py-2 bg-blue-600 hover:bg-blue-700 text-white rounded-lg text-sm disabled:opacity-50">
            {saving ? '저장 중...' : '저장'}
          </button>
        </Card>
      )}

      {templates.length === 0
        ? <EmptyState message="반복 전표 템플릿이 없습니다" icon="🔄" />
        : (
          <div className="space-y-2">
            {templates.map(t => (
              <Card key={t.id} className={`p-4 ${!t.is_active ? 'opacity-50' : ''}`}>
                <div className="flex items-center justify-between">
                  <div>
                    <p className="font-medium dark:text-white text-sm">{t.name}</p>
                    <p className="text-xs text-gray-500 mt-0.5">
                      매월 {t.day_of_month}일 · {t.accounts?.code} {t.accounts?.name} · {fmtKRW(t.amount)}
                    </p>
                  </div>
                  <div className="flex gap-2">
                    {t.is_active && (
                      <button onClick={() => handleGenerate(t)} disabled={saving}
                        className="text-xs px-2 py-1 bg-green-50 dark:bg-green-900/20 text-green-700 dark:text-green-400 border border-green-200 dark:border-green-800 rounded">
                        이번 달 초안
                      </button>
                    )}
                    <button onClick={() => toggleActive(t.id, !t.is_active)}
                      className="text-xs px-2 py-1 text-gray-500 border border-gray-200 dark:border-gray-700 rounded">
                      {t.is_active ? '중지' : '재개'}
                    </button>
                    <button onClick={() => deleteTemplate(t.id)}
                      className="text-xs px-2 py-1 text-red-400 border border-red-200 dark:border-red-800 rounded">
                      삭제
                    </button>
                  </div>
                </div>
              </Card>
            ))}
          </div>
        )
      }
    </div>
  )
}
```
