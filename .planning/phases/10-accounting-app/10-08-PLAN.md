```typescript
// 급여 계산 순수 함수 — 세율 로직 분리

export type IncomeType = 'business' | 'employment'

export interface PayrollResult {
  grossAmount: number
  withholdingTax: number
  netAmount: number
  incomeType: IncomeType
  effectiveRate: number  // 실효 세율 (%)
}

// 사업소득 (강사료): gross × 3.3% = 원천세 (소수점 버림)
// 지방소득세 포함: 3% (소득세) + 0.3% (지방소득세) = 3.3%
function calcBusinessIncome(gross: number): PayrollResult {
  const withholdingTax = Math.floor(gross * 0.033)
  return {
    grossAmount: gross,
    withholdingTax,
    netAmount: gross - withholdingTax,
    incomeType: 'business',
    effectiveRate: 3.3,
  }
}

// 근로소득: Phase 1 — 수동 입력 (간이세액표 자동화는 Phase 2)
function calcEmploymentIncome(gross: number, manualWithholding: number): PayrollResult {
  return {
    grossAmount: gross,
    withholdingTax: manualWithholding,
    netAmount: gross - manualWithholding,
    incomeType: 'employment',
    effectiveRate: gross > 0 ? (manualWithholding / gross) * 100 : 0,
  }
}

export function calcPayroll(
  gross: number,
  incomeType: IncomeType,
  manualWithholding = 0
): PayrollResult {
  if (gross <= 0) throw new Error('급여는 0보다 커야 합니다')
  if (incomeType === 'business') return calcBusinessIncome(gross)
  return calcEmploymentIncome(gross, manualWithholding)
}
```

```typescript
  // 폼 상태

  const [incomeType, setIncomeType] = useState<IncomeType>('business')
  const [grossStr, setGrossStr] = useState('')

  return (

          </p>
          <p className="text-xs text-yellow-600 dark:text-yellow-500 mt-0.5">
            ({thisMonthRecords.length}명 대상)
```

**src/views/reports/ArAgingView.tsx:**
```typescript
import { useState, useEffect } from 'react'
import { supabase } from '../../supabase'
import { Card, Spinner, EmptyState } from '../../components/shared/CommonUI'
import { fmtKRW } from '../../constants/theme'

interface AgingBucket { label: string; days: string; count: number; amount: number; color: string }
interface SyncSummary { last_sync: string; payment_count: number; new_entries: number }

interface Props { tenantId: string }

export default function ArAgingView({ tenantId }: Props) {
  const [buckets, setBuckets] = useState<AgingBucket[]>([])
  const [syncSummary, setSyncSummary] = useState<SyncSummary | null>(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    loadAgingData()
  }, [tenantId])

  const loadAgingData = async () => {
    setLoading(true)
    const today = new Date()

    // draft 전표 중 source_type='ryek_payment' 인 미확정 전표 (= 수납 대기)
    // entry_date 기준으로 경과 일수 계산
    const { data: draftPayments } = await supabase
      .from('journal_entries')
      .select('id, entry_date, description, journal_lines(amount, side, account_id, accounts(code))')
      .eq('tenant_id', tenantId)
      .eq('status', 'draft')
      .eq('source_type', 'ryek_payment')
      .order('entry_date', { ascending: true })

    // 구간별 집계
    const rawBuckets = [
      { label: '0–30일',  days: '0-30',  min: 0,  max: 30,  count: 0, amount: 0, color: 'text-green-600' },
      { label: '31–60일', days: '31-60', min: 31, max: 60,  count: 0, amount: 0, color: 'text-yellow-600' },
      { label: '61–90일', days: '61-90', min: 61, max: 90,  count: 0, amount: 0, color: 'text-orange-600' },
      { label: '90일 초과', days: '90+', min: 91, max: Infinity, count: 0, amount: 0, color: 'text-red-600' },
    ]

    for (const entry of draftPayments ?? []) {
      const entryDate = new Date(entry.entry_date)
      const diffDays = Math.floor((today.getTime() - entryDate.getTime()) / (1000 * 60 * 60 * 24))
      const lines = (entry.journal_lines ?? []) as { amount: number; side: string; accounts: { code: string } | null }[]
      const debitAmount = lines.filter(l => l.side === 'debit').reduce((s, l) => s + l.amount, 0)

      const bucket = rawBuckets.find(b => diffDays >= b.min && diffDays <= b.max)
      if (bucket) { bucket.count++; bucket.amount += debitAmount }
    }

    setBuckets(rawBuckets)

    // 최근 RYE-K 동기화 요약
    const { data: syncLog } = await supabase
      .from('ryek_sync_log')
      .select('synced_at, payment_count, new_entries')
      .eq('tenant_id', tenantId)
      .eq('status', 'success')
      .order('synced_at', { ascending: false })
      .limit(1)
      .single()

    if (syncLog) {
      setSyncSummary({
        last_sync: syncLog.synced_at,
        payment_count: syncLog.payment_count,
        new_entries: syncLog.new_entries,
      })
    }

    setLoading(false)
  }

  const totalCount  = buckets.reduce((s, b) => s + b.count, 0)
  const totalAmount = buckets.reduce((s, b) => s + b.amount, 0)

  return (
    <div className="p-4 max-w-2xl mx-auto">
      <h1 className="text-xl font-bold dark:text-white mb-4">미수금 현황 (AR Aging)</h1>

      {syncSummary && (
        <Card className="p-3 mb-4 bg-blue-50 dark:bg-blue-900/20 border-blue-200 dark:border-blue-800">
          <p className="text-xs text-blue-700 dark:text-blue-400">
            최근 RYE-K 동기화: {new Date(syncSummary.last_sync).toLocaleString('ko-KR')}
          </p>
          <p className="text-xs text-blue-600 dark:text-blue-300 mt-0.5">
            수납 완료 {syncSummary.payment_count}건 중 신규 전표 {syncSummary.new_entries}건 생성
          </p>
        </Card>
      )}

      {loading ? <Spinner /> : totalCount === 0 ? (
        <EmptyState message="미수금 전표가 없습니다. RYE-K 동기화 후 draft 전표를 확인하세요" icon="✅" />
      ) : (
        <>
          {/* 총계 */}
          <Card className="p-4 mb-4">
            <p className="text-xs text-gray-500 dark:text-gray-400">총 미수금</p>
            <p className="text-2xl font-bold text-ar dark:text-orange-400">{fmtKRW(totalAmount)}</p>
            <p className="text-xs text-gray-400 mt-1">{totalCount}건</p>
          </Card>

          {/* 구간별 막대 */}
          <div className="space-y-3">
```
