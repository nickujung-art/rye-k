```typescript
import { useState, useEffect } from 'react'
import { supabase } from '../../supabase'
import { Card, Spinner, EmptyState, FAB } from '../../components/shared/CommonUI'
import { fmtKRW, fmtDate } from '../../constants/theme'

interface Invoice {
  id: string
  institution_name: string
  biz_number: string | null
  invoice_date: string
  supply_amount: number
  description: string | null
  hometax_issued: boolean
  hometax_issued_at: string | null
  created_at: string
}

interface Props { tenantId: string }

// 홈택스 전자계산서 발행 URL
const HOMETAX_URL = 'https://www.hometax.go.kr/websquare/websquare.wss?w2xPath=/ui/pp/index_pp.xml&menuCd=PB0000000000000000000000000000000&tabGubun=TabType&nts=PP&lnkId=0101010000'

export default function InvoiceView({ tenantId }: Props) {
  const [invoices, setInvoices] = useState<Invoice[]>([])
  const [loading, setLoading] = useState(true)
  const [showForm, setShowForm] = useState(false)
  const [saving, setSaving] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const [confirmingId, setConfirmingId] = useState<string | null>(null)

  // 폼 상태
  const [institutionName, setInstitutionName] = useState('')
  const [bizNumber, setBizNumber]             = useState('')
  const [invoiceDate, setInvoiceDate]         = useState(new Date().toISOString().slice(0, 10))
  const [amountStr, setAmountStr]             = useState('')
  const [description, setDescription]         = useState('')

  const fetchInvoices = async () => {
    const { data } = await supabase
      .from('invoices')
      .select('*')
      .eq('tenant_id', tenantId)
      .order('invoice_date', { ascending: false })
      .limit(100)
    setInvoices((data ?? []) as Invoice[])
    setLoading(false)
  }

  useEffect(() => { fetchInvoices() }, [tenantId])

  const handleSave = async () => {
    const amount = parseInt(amountStr.replace(/,/g, ''), 10) || 0
    if (!institutionName.trim()) { setError('거래처명을 입력하세요'); return }
    if (amount <= 0) { setError('금액을 입력하세요'); return }

    setSaving(true)
    setError(null)
    try {
      await supabase.from('invoices').insert({
        tenant_id: tenantId,
        institution_name: institutionName.trim(),
        biz_number: bizNumber.trim() || null,
        invoice_date: invoiceDate,
        supply_amount: amount,
        description: description.trim() || null,
        hometax_issued: false,
      })
      setShowForm(false)
      setInstitutionName(''); setBizNumber(''); setAmountStr(''); setDescription('')
      fetchInvoices()
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : '저장 실패')
    } finally {
      setSaving(false)
    }
  }

  const handleHometaxClick = (invoice: Invoice) => {
    // 홈택스 새 탭으로 열기
    window.open(HOMETAX_URL, '_blank', 'noopener,noreferrer')
    // 발행 완료 확인 UI 표시
    setConfirmingId(invoice.id)
  }

  const handleIssueConfirm = async (invoiceId: string) => {
    await supabase.from('invoices').update({
      hometax_issued: true,
      hometax_issued_at: new Date().toISOString(),
    }).eq('id', invoiceId)
    setConfirmingId(null)
    fetchInvoices()
  }

  if (loading) return <Spinner />

  return (
    <div className="p-4 max-w-2xl mx-auto">
      <div className="flex items-center justify-between mb-4">
        <h1 className="text-xl font-bold dark:text-white">B2B 계산서</h1>
        <button onClick={() => setShowForm(s => !s)}
          className="px-3 py-1.5 bg-blue-600 text-white text-sm rounded-lg hover:bg-blue-700">
          + 계산서 작성
        </button>
      </div>

      {/* 작성 폼 */}
      {showForm && (
        <Card className="p-4 mb-4 space-y-3">
          <h2 className="text-sm font-semibold dark:text-white">계산서 초안 작성</h2>

          <div className="grid grid-cols-2 gap-2">
            <div>
              <label className="text-xs text-gray-500 block mb-1">거래처명 *</label>
              <input type="text" value={institutionName} onChange={e => setInstitutionName(e.target.value)}
                placeholder="예: ○○학교"
                className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm" />
            </div>
            <div>
              <label className="text-xs text-gray-500 block mb-1">사업자등록번호</label>
              <input type="text" value={bizNumber} onChange={e => setBizNumber(e.target.value)}
                placeholder="000-00-00000"
                className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm" />
            </div>
          </div>

          <div className="grid grid-cols-2 gap-2">
            <div>
              <label className="text-xs text-gray-500 block mb-1">작성일</label>
              <input type="date" value={invoiceDate} onChange={e => setInvoiceDate(e.target.value)}
                className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm" />
            </div>
            <div>
              <label className="text-xs text-gray-500 block mb-1">공급가액 *</label>
              <input type="text" inputMode="numeric" value={amountStr}
                onChange={e => { const r = e.target.value.replace(/\D/g, ''); setAmountStr(r ? parseInt(r).toLocaleString('ko-KR') : '') }}
                placeholder="0"
                className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm text-right" />
            </div>
          </div>

          <div>
            <label className="text-xs text-gray-500 block mb-1">품목/설명</label>
            <input type="text" value={description} onChange={e => setDescription(e.target.value)}
              placeholder="예: 6월 국악 수업비"
              className="w-full px-3 py-2 border rounded-lg dark:bg-gray-700 dark:border-gray-600 dark:text-white text-sm" />
          </div>

          {/* 부가세 면세 안내 */}
          <div className="bg-blue-50 dark:bg-blue-900/20 rounded-lg p-3">
            <p className="text-xs text-blue-700 dark:text-blue-400 font-medium">학원 부가세 면세 사업자 안내</p>
            <p className="text-xs text-blue-600 dark:text-blue-300 mt-1">
              학원 수강료는 부가세 면세 항목입니다. 홈택스에서 전자계산서(면세) 로 발행하세요.
            </p>
          </div>

          {error && <p className="text-sm text-red-500">{error}</p>}

          <button onClick={handleSave} disabled={saving}
            className="w-full py-2.5 bg-blue-600 hover:bg-blue-700 disabled:bg-blue-300
                       text-white font-medium rounded-lg transition-colors">
            {saving ? '저장 중...' : '초안 저장'}
          </button>
        </Card>
      )}

      {/* 발행 확인 인라인 UI */}
      {confirmingId && (
        <Card className="p-4 mb-4 border-green-200 dark:border-green-800 bg-green-50 dark:bg-green-900/20">
          <p className="text-sm font-medium text-green-800 dark:text-green-300 mb-3">
            홈택스에서 발행이 완료되었나요?
          </p>
          <div className="flex gap-2">
            <button onClick={() => handleIssueConfirm(confirmingId)}
              className="flex-1 py-2 bg-green-600 hover:bg-green-700 text-white text-sm rounded-lg">
              발행 완료 확인
            </button>
            <button onClick={() => setConfirmingId(null)}
              className="flex-1 py-2 border border-gray-300 dark:border-gray-600 text-gray-600 dark:text-gray-400 text-sm rounded-lg">
              아직 미발행
            </button>
          </div>
        </Card>
      )}

      {/* 계산서 목록 */}
      {invoices.length === 0
        ? <EmptyState message="발행된 계산서가 없습니다" icon="📋" />
        : (
          <div className="space-y-2">
            {invoices.map(inv => (
              <Card key={inv.id} className="p-4">
                <div className="flex items-start justify-between">
                  <div className="flex-1">
                    <div className="flex items-center gap-2 mb-1">
                      <span className={`text-xs px-1.5 py-0.5 rounded-full font-medium ${
                        inv.hometax_issued
                          ? 'bg-green-100 dark:bg-green-900/30 text-green-700 dark:text-green-400'
                          : 'bg-yellow-100 dark:bg-yellow-900/30 text-yellow-700 dark:text-yellow-400'
                      }`}>
                        {inv.hometax_issued ? '발행완료' : '미발행'}
                      </span>
                      <span className="text-xs text-gray-400">{fmtDate(inv.invoice_date)}</span>
                    </div>
                    <p className="font-medium dark:text-white text-sm">{inv.institution_name}</p>
                    {inv.description && (
                      <p className="text-xs text-gray-500 mt-0.5">{inv.description}</p>
                    )}
                    <p className="text-sm font-semibold text-blue-600 dark:text-blue-400 mt-1">
                      {fmtKRW(inv.supply_amount)}
                    </p>
                    {inv.hometax_issued && inv.hometax_issued_at && (
                      <p className="text-xs text-gray-400 mt-0.5">
                        발행일: {new Date(inv.hometax_issued_at).toLocaleDateString('ko-KR')}
                      </p>
                    )}
                  </div>

                  {!inv.hometax_issued && (
                    <button onClick={() => handleHometaxClick(inv)}
                      className="ml-3 px-3 py-1.5 bg-orange-500 hover:bg-orange-600 text-white text-xs rounded-lg font-medium whitespace-nowrap">
                      홈택스 발행
                    </button>
                  )}
                </div>
              </Card>
            ))}
          </div>
        )
      }
    </div>
  )
}
```
